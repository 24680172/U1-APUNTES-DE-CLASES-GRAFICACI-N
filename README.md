    Limpia la escena y genera un patr√≥n de c√≠rculos conc√©ntricos.
    """
    # 1. Limpiar escena (Borra todo el contenido previo)
    bpy.ops.object.select_all(action='SELECT')
    bpy.ops.object.delete()

    # 2. C√≠rculo Central
    bpy.ops.mesh.primitive_circle_add(
        radius=radio, 
        location=(0, 0, 0), 
        vertices=64
    )

    # 3. Generaci√≥n de c√≠rculos perif√©ricos mediante Bucle While
    angulo_actual = 0
    
    while angulo_actual < 360:
        # Convertimos grados a radianes para las funciones math.cos y math.sin
        radianes = math.radians(angulo_actual)
        
        # Calcular coordenadas (x, y) usando trigonometr√≠a
        x = radio * math.cos(radianes)
        y = radio * math.sin(radianes)
        
        # Crear el c√≠rculo en la posici√≥n calculada
        bpy.ops.mesh.primitive_circle_add(
            radius=radio, 
            location=(x, y, 0), 
            vertices=64
        )
        
        # Incrementar el √°ngulo para la siguiente iteraci√≥n
        angulo_actual += paso_angular

# Ejecutar la funci√≥n
# Tip: Cambia paso_angular a 60 para 6 p√©talos o a 30 para 12.
generar_patron_circular(radio=3, paso_angular=60)
